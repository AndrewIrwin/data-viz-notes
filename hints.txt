# incomplete and crappy, but it helps.
getLessons <- function(directory = "lessons") {
  fl_qmd <- list.files(here::here(directory), ".Rmd", full.names=TRUE)
  lessons <- lapply(fl_qmd, readLines) |> unlist()
  pkgs0 <- sort(unique(lessons[grep("library\\(", lessons)]))
  needed <- str_remove(pkgs0, "^library") |> str_remove_all("\\(|\\)") |> str_squish()
  have <- installed.packages()[, "Package"]
  setdiff(needed, have)
}
